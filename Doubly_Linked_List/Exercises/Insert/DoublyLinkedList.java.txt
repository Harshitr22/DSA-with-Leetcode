public class DoublyLinkedList {
    private Node head;
    private Node tail;
    private int length;

    class Node{
        int value;
        Node next;
        Node prev;

        Node(int value){
            this.value = value;
        }
    }

    DoublyLinkedList(int value){
        Node newNode = new Node(value);
        head = newNode;
        tail = newNode;
        length = 1;
    }

        System.out.println("Head: " + head.value);
    }

    public void append(int value){

        if(length == 0){

            tail = newNode;
        }else{
            tail.next = newNode;
            newNode.prev = tail;

        }
        length++;
