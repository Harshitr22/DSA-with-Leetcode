public class Main {
    public static void main(String[] args){
        LinkedList linkedList = new LinkedList(1);

        System.out.println("Before Appending a value: ");
        linkedList.getHead();
        linkedList.getTail();
        linkedList.getLength();
        linkedList.printList();
        System.out.println();

        System.out.println("After Appending a value: ");
        linkedList.append(5);
        linkedList.append(7);
        linkedList.append(10);
        linkedList.append(4);

	// Testing of the Swapping Nodes in pair
        System.out.println("Swapping the Nodes on pair:");
        linkedList.printList();
        linkedList.swapPairs();
        linkedList.printList();
        linkedList.getHead();
        System.out.println();

    }

}
