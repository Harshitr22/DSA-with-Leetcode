ublic class LinkedList {
    private Node head;
    private Node tail;
    private int length;

    class Node{
        int value;
        Node next;

        Node(int value) {
            this.value = value;
        }
    }

    LinkedList(int value){
        Node newNode = new Node(value);
        head = newNode;
        tail = newNode;
        length = 1;
    }

    public void getHead(){
        System.out.println("Head: " + head.value);
    }

    public void getTail(){
        System.out.println("Tail: " + tail.value);
    }

    public void getLength(){
        System.out.println("Length: " + length);
    }

    public void printList(){
        Node temp = head;
        System.out.print("The Items in List are: ");
        while(temp != null){
            System.out.print( temp.value + " -> ");
            temp = temp.next;
        }
        System.out.print("null");
        System.out.println();
    }

    // Partition List
    public void partitionList(int value){
        Node dummyNode1 = new Node(0);
        Node dummyNode2 = new Node(0);
        Node prev1 = dummyNode1;
        Node prev2 = dummyNode2;
        Node current = head;
        while(current != null){
            if(current.value < value){
                prev1.next = current;
                current = current.next;
                prev1 = prev1.next;
                prev1.next = null;
            } else{
                prev2.next = current;
                current = current.next;
                prev2 = prev2.next;
                prev2.next = null;
            }
        }
        prev1.next = dummyNode2.next;
        head = dummyNode1.next;
    }

}
