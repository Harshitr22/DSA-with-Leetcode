public class Queue {
    private Node first;
    private Node last;
    private int length;

    Queue(int value){
        Node newNode = new Node(value);
        first = newNode;
        last = newNode;
        length = 1;
    }

    class Node{
        int value;
        Node next;

        Node(int value){
            this.value = value;
        }
    }

    public void getFirst(){
        System.out.println("The First in the Queue is: " + first.value);
    }

    public void getLast(){
        System.out.println("The Last in the Queue is: " + last.value);
    }

    public void getLength(){
        System.out.println("The Length of the Queue is: " + length);
    }

    public void printQueue(){
        Node temp = first;
        System.out.print("The Items in List are: ");
        while(temp != null){
            System.out.print( temp.value + " -> ");
            temp = temp.next;
        }
        System.out.print("null");
        System.out.println();
    }

    // Method to add an element in the queue
    public void enqueue(int value){
        Node newNode = new Node(value);
        if(length == 0){
            first = newNode;
            last = newNode;
        }else{
            last.next = newNode;
            last = newNode;
        }
        length++;
    }

    //Method to remove an element from the queue, the element will always be removed from the front/first
    public Node dequeue(){
        if(length == 0) return null;
        Node temp = first;
        if(length == 1){
            first = null;
            last = null;
        }else{
            first = first.next;
            temp.next = null;
        }
        length--;
       return temp;
    }

}
